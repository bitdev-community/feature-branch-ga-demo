import { ThemeProvider } from '@showoff/design.theme.theme-provider';

A component that renders the contact artwork based on theme.

### Component usage

Use with `ThemeProvider` to use the correct colors.

```js
<Artwork />
```

### Render hello world!

```js live
<ThemeProvider theme={lightTheme()}>
  <Artwork />
</ThemeProvider>
```
